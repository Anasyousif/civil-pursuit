/**
 * Return the items being currently evaluated in an evaluation
 * 
 * @module synapp
 * @class Filter
 * @method filter::currently-evaluated
 * @param {Array} items - Items to parse
 * @example
 *    <!-- HTML -->
 *    <ANY ng-repeat='item in items | currentlyEvaluatedFilter' />
 *    
 *    // JS
 *    var currentlyEvaluated = currentlyEvaluatedFilter(items);
 * @author francoisrvespa@gmail.com
*/

module.exports = function () {
  return function (items) {

    var current = [];
    
    if ( Array.isArray(items) && items.length ) {
      
      current.push(items[0]);

      if ( items[1] ) {
        current.push(items[1]);
      }
    }

    return current;
  };
};
